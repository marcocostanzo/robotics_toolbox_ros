176  /*
177  UnitQuaternion.interp Interpolate UnitQuaternions
178 
179  QI = Q.scale(S, OPTIONS) is a UnitQuaternion that interpolates between a null
180  rotation (identity quaternion) for S=0 to Q for S=1.
181 
182  QI = Q.interp(Q2, S, OPTIONS) as above but interpolates a rotation
183  between Q for S=0 and Q2 for S=1.
184 
185  Options::
186  'shortest' Take the shortest path along the great circle
187 
188  Notes::
189  - This is a spherical linear interpolation (slerp) that can be interpretted
190  as interpolation along a great circle arc on a sphere.
191  - It is an error if S is outside the interval 0 to 1.
192 
193  References::
194  - Animating rotation with quaternion curves,
195  K. Shoemake,
196  in Proceedings of ACM SIGGRAPH, (San Francisco), pp. 245-254, 1985.
197 
198  See also ctraj.
199  */
200  UnitQuaternion UnitQuaternion::interp(const UnitQuaternion& q2, double s, bool b_shortest) const {
201 
202  Vector<4> q1_vec = getDouble();
203  Vector<4> q2_vec = q2.getDouble();
204 
205  double cosTheta = q1_vec*q2_vec;
206 
207  if( b_shortest ){
208  //take shortest path along the great circle, patch by Gauthier Gras
209  if( cosTheta < 0 ){
210  q1_vec = - q1_vec;
211  cosTheta = - cosTheta;
212  }
213  }
214 
215  double theta = acos(cosTheta);
216 
217  //check s range
218  if(s<0 || s>1){
219  cout << PORTING_FUNCTIONS_FAIL_COLOR "[UnitQuaternion]: error in interp() | s must be in [0 1] | s = " << s << PORTING_FUNCTIONS_CRESET << endl;
220  exit(-1);
221  }
222 
223  if(theta == 0)
224  return UnitQuaternion(q1_vec);
225 
226  return UnitQuaternion( (sin((1-s)*theta) * q1_vec + sin(s*theta) * q2_vec) / sin(theta) );
227 
228  }

551  /*
552  TOROT Convert UnitQuaternion to homogeneous transform
553 
554  T = q2tr(Q)
555 
556  Return the rotational homogeneous transform corresponding to the unit
557  quaternion Q.
558 
559  See also: TR2Q
560  */
561  Matrix<3,3> UnitQuaternion::torot() const{
562 
563  Vector<4> q_vec = getDouble();
564  double s = q_vec[0];
565  double x = q_vec[1];
566  double y = q_vec[2];
567  double z = q_vec[3];
568 
569  return Data(
570 
571  1.0-2.0*( pow(y,2)+pow(z,2) ), 2.0*( x*y-s*z ), 2.0*( x*z+s*y ),
572  2.0*( x*y+s*z ), 1.0-2.0*( pow(x,2)+pow(z,2) ), 2.0*( y*z-s*x ),
573  2.0*( x*z-s*y ), 2.0*( y*z+s*x ), 1.0-2.0*( pow(x,2)+pow(y,2) )
574 
575  );
576  }

708  /*
709  TR2Q Convert homogeneous transform to a UnitQuaternion
710 
711  Q = tr2q(T)
712 
713  Return a UnitQuaternion corresponding to the rotational part of the
714  homogeneous transform T.
715 
716  Reference::
717  - Funda, Taylor, IEEE Trans. Robotics and Automation, 6(3), June 1990, pp.
718  */
719  UnitQuaternion UnitQuaternion::r2q( const Matrix<3>& R ){
720 
721  //Check matrix
722  if( !isRot(R) ){
723  throw std::invalid_argument("[UnitQuaternion] Error in r2q( const Matrix<3,3>& R ): R is not a rotation Matrix");
724  }
725 
726  //Using modified version of sign() function as per the paper
727  // sign(x) = 1 if x>=0
728 
729  double trace_1 = trace(R)+1.0;
730  if( trace_1 < 0.0 ){
731  if( trace_1 > -10.0*std::numeric_limits<double>::epsilon() ){
732  trace_1 = 0.0; //<- Numeric fix
733  } else {
734  // trace+1 is negative for another reason
735  throw std::logic_error("[UnitQuaternion] Error in r2q( const Matrix<3,3>& R ): trace+1 of R is negative. Please contact the mantainer.");
736  }
737  }
738  double s = sqrt(trace_1)/2.0;
739  double kx = R[2][1] - R[1][2]; // Oz - Ay
740  double ky = R[0][2] - R[2][0]; // Ax - Nz
741  double kz = R[1][0] - R[0][1]; // Ny - Ox
742 
743  // for the numerical case, deal with rotations by very small angles
744 
745  // equation (7)
746  int k = index_max(R.diagonal_slice());
747  double kx1,ky1,kz1;
748  double sgn;
749  switch (k)
750  {
751  case 0:{ // Nx dominates
752  kx1 = R[0][0] - R[1][1] - R[2][2] + 1.0; // Nx - Oy - Az + 1
753  ky1 = R[1][0] + R[0][1]; // Ny + Ox
754  kz1 = R[2][0] + R[0][2]; // Nz + Ax
755  sgn = sign(kx, 1);
756  break;
757  }
758  case 1:{ // Oy dominates
759  kx1 = R[1][0] + R[0][1]; // Ny + Ox
760  ky1 = R[1][1] - R[0][0] - R[2][2] + 1.0; // Oy - Nx - Az + 1
761  kz1 = R[2][1] + R[1][2]; // Oz + Ay
762  sgn = sign(ky, 1);
763  break;
764  }
765  case 2:{ // Az dominates
766  kx1 = R[2][0] + R[0][2]; // Nz + Ax
767  ky1 = R[2][1] + R[1][2]; // Oz + Ay
768  kz1 = R[2][2] - R[0][0] - R[1][1] + 1.0; // Az - Nx - Oy + 1
769  sgn = sign(kz, 1);
770  break;
771  }
772  default:{
773  /* Default Code */
774  throw std::logic_error("[UnitQuaternion] Error in r2q( const Matrix<3,3>& R ): invalid index in index_max(). Please contact the mantainer.");
775  }
776  }
777 
778  // equation (8)
779  kx = kx + sgn*kx1;
780  ky = ky + sgn*ky1;
781  kz = kz + sgn*kz1;
782 
783  Vector<3> k_vec = makeVector(kx, ky, kz);
784  double nm = TooN::norm( k_vec );
785  if(nm == 0){
786  //handle special case of null quaternion
787  return UnitQuaternion();
788  } else {
789  return UnitQuaternion(s , k_vec * sqrt( 1.0 - pow(s,2) ) / nm );
790  }
791 
792  }
