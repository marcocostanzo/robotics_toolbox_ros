116 Vector<3> r2rpy( const Matrix<3,3>& R ){
117 
118  //Check matrix
119  if( !isRot(R) ){
120  throw std::invalid_argument("[PORTINGFUNCIONS] Error in r2rpy( const Matrix<3,3>& R ): R is not a rotation Matrix");
121  }
122 
123  //old ZYX order (as per Paul book)
124 
125  Vector<3> rpy = Zeros;
126 
127  if( fabs( fabs(R[2][0]) - 1.0 ) < GEOMETRY_HELPER_EPSILON ){ //when |R31| == 1
128  //singularity
129 
130  rpy[0] = 0; // roll is zero
131  if( R[2][0] < 0.0 ){
132  rpy[2] = -atan2( R[0][1] , R[0][2] ); // R-Y
133  } else {
134  rpy[2] = atan2( -R[0][1], -R[0][2] ); // R+Y
135  }
136  rpy[1] = -asin( R[2][0] );
137  } else {
138  rpy[0] = atan2( R[2][1], R[2][2] ); // R
139  rpy[2] = atan2( R[1][0], R[0][0] ); // Y
140 
141  rpy[1] = -atan( R[2][0] * cos( rpy[0] ) / R[2][2] );
142  }
143 
144  return rpy;
145 
146 }

166 Vector<3> r2eul( const Matrix<3,3>& R ){
167 
168  //Check matrix
169  if( !isRot(R) ){
170  throw std::invalid_argument("[PORTINGFUNCIONS] Error in r2eul( const Matrix<3,3>& R ): R is not a rotation Matrix");
171  }
172 
173  /* Method as per Paul, p 69.
174  euler = [phi theta psi]
175  */
176 
177  Vector<3> eul = Zeros;
178 
179  if( abs( R[0][2] ) < GEOMETRY_HELPER_EPSILON && abs( R[1][2] ) < GEOMETRY_HELPER_EPSILON ){
180  //singularity
181  eul[0] = 0.0;
182  double sp = 0.0;
183  double cp = 0.0;
184  eul[1] = atan2( cp*R[0][2] + sp*R[1][2], R[2][2]);
185  eul[2] = atan2( -sp * R[0][0] + cp * R[1][0], -sp*R[0][1] + cp*R[1][1] );
186  } else {
187  // non singular
188  eul[0] = atan2( R[1][2] , R[0][2] );
189  double sp = sin(eul[0]);
190  double cp = cos(eul[0]);
191  eul[1] = atan2( cp*R[0][2] + sp*R[1][2] , R[2][2] );
192  eul[2] = atan2(-sp * R[0][0] + cp * R[1][0], -sp*R[0][1] + cp*R[1][1]);
193  }
194 
195  return eul;
196 
197 }
