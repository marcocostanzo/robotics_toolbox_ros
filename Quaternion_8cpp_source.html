180  /*
181  Quaternion.matrix Matrix representation of Quaternion
182 
183  M = Q.matrix() is a matrix (4x4) representation of the Quaternion Q.
184 
185  Quaternion, or Hamilton, multiplication can be implemented as a
186  matrix-vector product, where the column-vector is the elements of a
187  second quaternion:
188 
189  matrix(Q1) * double(Q2)'
190 
191  Notes::
192  - This matrix is not unique, other matrices will serve the purpose for
193  multiplication, see https://en.wikipedia.org/wiki/Quaternion#Matrix_representations
194  - The determinant of the matrix is the norm of the quaternion to the fourth power.
195 
196  See also Quaternion.double, Quaternion.mtimes.
197  */
198  Matrix<4,4> Quaternion::matrix() const{
199  /*
200  Matlab Version:
201  m = [q.s -q.v(1) -q.v(2) -q.v(3)
202  q.v(1) q.s -q.v(3) q.v(2)
203  q.v(2) q.v(3) q.s -q.v(1)
204  q.v(3) -q.v(2) q.v(1) q.s];
205  */
206  return Data( _s, -_v[0], -_v[1], -_v[2],
207  _v[0], _s, -_v[2], _v[1],
208  _v[1], _v[2], _s, -_v[0],
209  _v[2], -_v[1], _v[0], _s );
210  }

233  /*
234  Quaternion.mtimes Multiply a quaternion object
235 
236  Q1*Q2 is a quaternion formed by the Hamilton product of two quaternions.
237  Q*S is the element-wise multiplication of quaternion elements by the scalar S.
238  S*Q is the element-wise multiplication of quaternion elements by the scalar S.
239 
240  Notes::
241  - Overloaded operator '*'
242 
243  See also Quaternion.mrdivide, Quaternion.mpower.
244  */
245  Quaternion Quaternion::mtimes(const Quaternion& q2) const{
246  double s1 = _s;
247  double s2 = q2.getS();
248  Vector<3> v1 = _v;
249  Vector<3> v2 = q2.getV();
250  return Quaternion( s1*s2 - (v1*v2) , s1*v2 + s2*v1 + (v1^v2) );
251  }

288  /*
289  Quaternion.mpower Raise quaternion to integer power
290 
291  Q^N is the Quaternion Q raised to the integer power N.
292 
293  Notes::
294  - Overloaded operator '^'
295  - Computed by repeated multiplication.
296  - If the argument is a unit-quaternion, the result will be a
297  unit quaternion.
298 
299  See also Quaternion.mtimes.
300  */
301  Quaternion Quaternion::mpower(int p) const{
302 
303  if( p == 0 )
304  return Quaternion(1.0, Zeros);
305 
306  //multiply by itself so many times
307  Quaternion q(*this);
308  for( int i=1; i<abs(p); i++ ){
309  q = q * (*this);
310  }
311 
312  // if exponent was negative, invert it
313  if( p<0 ){
314  q = ::inv(q);
315  }
316 
317  return q;
318 
319  }
320  Quaternion Quaternion::operator^(const int& p) const{
321  return mpower(p);
322  }
